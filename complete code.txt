package aman;

import java.util.ArrayList; 
import java.util.List;
import java.util.Scanner;

abstract class User { 
	//base class
    protected String registrationNumber; // field for registration no.
    protected String password; // field for password

    public User(String registrationNumber, String password) {
        this.registrationNumber = registrationNumber;
        this.password = password;
    }

    public abstract void displayRole();
}

class RegularUser extends User {
    private List<String> quizHistory = new ArrayList<>(); //field for quiz history

    public RegularUser(String registrationNumber, String password) {
    	// Constructor for RegularUser
        super(registrationNumber, password);
    }

    @Override
    public void displayRole() {
        System.out.println("Role: Regular User");
    }

    public void viewProfile() {
        System.out.println("Viewing profile for " + registrationNumber);
    }

    public void addQuizToHistory(String quizName) { //parameter
        quizHistory.add(quizName);
    }

    public void showQuizHistory() {
        System.out.println("Quiz History: " + quizHistory);
    }
}

class Quiz {  
	// High cohesion class
    private String category;
    private String difficultyLevel;
    private List<String> questions;
    private List<String[]> options;
    private List<String> correctAnswers;

    public Quiz(String category, String difficultyLevel) {
    	// Constructor for Quiz
        this.category = category;
        this.difficultyLevel = difficultyLevel;
        this.questions = new ArrayList<>();
        this.options = new ArrayList<>();
        this.correctAnswers = new ArrayList<>();
    }
     //accessor
    public String getCategory() {
        return category;
    }

    public String getDifficultyLevel() {
        return difficultyLevel;
    }

    //mutator
    public void addQuestion(String question, String[] options, String correctAnswer) {
        questions.add(question);
        this.options.add(options);
        correctAnswers.add(correctAnswer);
    }

    public List<String> getQuestions() {
        return questions;
    }

    public String[] getOptions(int index) {
        return options.get(index);
    }

    public String getCorrectAnswer(int index) {
        return correctAnswers.get(index);
    }

    public void displayQuiz() {
        System.out.println("Category: " + category);
        System.out.println("Difficulty: " + difficultyLevel);
    }
}

    private static void startQuiz(Quiz quiz, RegularUser user, Scanner sc) {//parameter
        int score = 0;
        List<String> incorrectAnswers = new ArrayList<>();
        try {
            for (int i = 0; i < quiz.getQuestions().size(); i++) {
                System.out.println("\n\n" + (i + 1) + ". " + quiz.getQuestions().get(i));

                // Defensive programming for options retrieval
                if (quiz.getOptions(i) == null || quiz.getOptions(i).length == 0) {
                    System.out.println("Error: No options available for this question.");
                    continue; // Skip this question
                }

                String[] opts = quiz.getOptions(i);
                for (int j = 0; j < opts.length; j++) {
                    System.out.println("   " + (char) ('A' + j) + ". " + opts[j]);
                }

                System.out.print("Your Answer (A/B/C/D): ");
                String answer = sc.nextLine().toUpperCase();

                // Validate user input
                if (!answer.matches("[A-D]")) {
                    System.out.println("Invalid input. Please enter A, B, C, or D.");
                    i--; // Repeat the question
                    continue;
                }

                String correctAnswer = quiz.getCorrectAnswer(i);
                if (correctAnswer == null) {
                    System.out.println("Error: No correct answer found for this question.");
                    continue; // Skip this question
                }

                if (answer.equals(correctAnswer)) {
                    score++;
                } else {
                    incorrectAnswers.add((i + 1) + ". " + quiz.getQuestions().get(i));
                    incorrectAnswers.add("Correct Answer: " + correctAnswer);
                }
            }
        } catch (IndexOutOfBoundsException e) {
            System.out.println("Error: Question or answer index is out of bounds. " + e.getMessage());
        } catch (Exception e) {
            System.out.println("An unexpected error occurred during the quiz: " + e.getMessage());
        }

        System.out.println("\nQuiz Completed!");
        System.out.println("Your Score: " + score + "/" + quiz.getQuestions().size());
        user.addQuizToHistory(quiz.getCategory() + " Quiz (" + quiz.getDifficultyLevel() + ")");
        user.showQuizHistory();

        if (!incorrectAnswers.isEmpty()) {
            System.out.println("\nIncorrect Answers:");
            for (String incorrectAnswer : incorrectAnswers) {
                System.out.println(incorrectAnswer);
            }
        } else {
            System.out.println("All answers were correct!");
        }
    }
